Database commands

create database rideShare;

use rideShare;

create table driver (
uName varchar (50) not null unique, 
fName varchar (50) not null,
lName varchar (50) not null,
smoker boolean, 
gender binary not null,
pet binary,
cargo binary,

primary key(uName)
);

create table passanger (
uName varchar (50) not null unique, 
fName varchar (50) not null,
lName varchar (50) not null,
smoker boolean, 
gender binary not null,

primary key(uName)
);
